CREATE TABLE Admins (
  ID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
  Name VARCHAR(25),
  StreetAddress VARCHAR(25),
  City VARCHAR(14),
  State VARCHAR(2),
  ZipCode VARCHAR(5)
);

CREATE TABLE Members (
  ID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
  Name VARCHAR(25),
  StreetAddress VARCHAR(25),
  City VARCHAR(14),
  State VARCHAR(2),
  ZipCode VARCHAR(5),
  Status SHORT
);

CREATE TABLE Managers (
  ID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
  Name VARCHAR(25),
  StreetAddress VARCHAR(25),
  City VARCHAR(14),
  State VARCHAR(2),
  ZipCode VARCHAR(5)
);

CREATE TABLE Providers (
  ID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
  Name VARCHAR(25),
  StreetAddress VARCHAR(25),
  City VARCHAR(14),
  State VARCHAR(2),
  ZipCode VARCHAR(5),
  BankAccount BIGINT
);

CREATE TABLE ReportSchedules (
  ID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
  ReportType INTEGER NOT NULL,
  WeekDay SHORT,
  Time TIME
);

CREATE TABLE Services (
  ID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
  ServiceCode INTEGER NOT NULL,
  Name VARCHAR(25),
  Fee DECIMAL(3,2)
);

CREATE TABLE ServiceRecords (
  ID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
  ServiceCode INTEGER,
  ProviderID INTEGER,
  MemberID INTEGER,
  Comments VARCHAR(100),
  DateProvided DATETIME,
  TimeStamp DATETIME,
  FOREIGN KEY(ServiceCode) REFERENCES Services(ServiceCode),
  FOREIGN KEY(ProviderID) REFERENCES Providers(ID),
  FOREIGN KEY(MemberID) REFERENCES Members(ID)
);
